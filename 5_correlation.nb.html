get_data <- function(mob_data, rt_data, lock_start, lock_end, cols) {
  df <- mob_data %>% 
    left_join(rt_data, by=c("date" = "Date")) %>% 
    select(all_of(cols)) %>% 
    mutate(date = as.Date(date)) %>% 
    rename(retail=retail_and_recreation_percent_change_from_baseline,grocery=grocery_and_pharmacy_percent_change_from_baseline, parks=parks_percent_change_from_baseline, transit=transit_stations_percent_change_from_baseline, workplaces=workplaces_percent_change_from_baseline, residential=residential_percent_change_from_baseline, global=`Global Mobility Score`) %>% 
    mutate(retail_lag7 = lag(retail, 7)) %>% 
    mutate(grocery_lag7 = lag(grocery, 7) ) %>% 
    mutate(parks_lag7 = lag(parks, 7)) %>%
    mutate(transit_lag7 = lag(transit, 7)) %>%
    mutate(workplaces_lag7 = lag(workplaces, 7)) %>%
    mutate(residential_lag7 = lag(residential, 7)) %>%
    mutate(global_lag7 = lag(global, 7)) %>%
    mutate(retail_lag14 = lag(retail, 14)) %>% 
    mutate(grocery_lag14 = lag(grocery, 14) ) %>% 
    mutate(parks_lag14 = lag(parks, 14)) %>%
    mutate(transit_lag14 = lag(transit, 14)) %>%
    mutate(workplaces_lag14 = lag(workplaces, 14)) %>%
    mutate(residential_lag14 = lag(residential, 14)) %>%
    mutate(global_lag14 = lag(global, 14)) %>%
    filter(date>=lock_start & date<=lock_end)
  return(df)
}

Function to get correlation data

Input: Df of specific region, desired mobility lag time (0, 7, or 14 days)
Output: Df of specific region with list of correlations for each mobility

get_correlations <- function(data, lag) {
  if(lag==0) {
    retail <- pcor.test(data$retail, data$Rt, as.numeric(data$date), method = 'pearson')
    grocery <- pcor.test(data$grocery, data$Rt, as.numeric(data$date), method = 'pearson')
    parks <- pcor.test(data$parks, data$Rt, as.numeric(data$date), method = 'pearson')
    transit <- pcor.test(data$transit, data$Rt, as.numeric(data$date), method = 'pearson')
    workplaces <- pcor.test(data$workplaces, data$Rt, as.numeric(data$date), method = 'pearson')
    residential <- pcor.test(data$residential, data$Rt, as.numeric(data$date), method = 'pearson')
    global <- pcor.test(data$global, data$Rt, as.numeric(data$date), method = 'pearson')
  } else if(lag==7) {
    retail <- pcor.test(data$retail_lag7, data$Rt, as.numeric(data$date), method = 'pearson')
    grocery <- pcor.test(data$grocery_lag7, data$Rt, as.numeric(data$date), method = 'pearson')
    parks <- pcor.test(data$parks_lag7, data$Rt, as.numeric(data$date), method = 'pearson')
    transit <- pcor.test(data$transit_lag7, data$Rt, as.numeric(data$date), method = 'pearson')
    workplaces <- pcor.test(data$workplaces_lag7, data$Rt, as.numeric(data$date), method = 'pearson')
    residential <- pcor.test(data$residential_lag7, data$Rt, as.numeric(data$date), method = 'pearson')
    global <- pcor.test(data$global_lag7, data$Rt, as.numeric(data$date), method = 'pearson')
  } else if(lag==14) {
    retail <- pcor.test(data$retail_lag14, data$Rt, as.numeric(data$date), method = 'pearson')
    grocery <- pcor.test(data$grocery_lag14, data$Rt, as.numeric(data$date), method = 'pearson')
    parks <- pcor.test(data$parks_lag14, data$Rt, as.numeric(data$date), method = 'pearson')
    transit <- pcor.test(data$transit_lag14, data$Rt, as.numeric(data$date), method = 'pearson')
    workplaces <- pcor.test(data$workplaces_lag14, data$Rt, as.numeric(data$date), method = 'pearson')
    residential <- pcor.test(data$residential_lag14, data$Rt, as.numeric(data$date), method = 'pearson')
    global <- pcor.test(data$global_lag14, data$Rt, as.numeric(data$date), method = 'pearson')
  } else {
    stop('Must enter 0, 7, or 14')
  }

  return(list(Retail=retail, Grocery=grocery, Parks=parks, Transit=transit, Workplaces=workplaces, Residential=residential, Global=global))
}

Function to get correlation coefficient

Input: List of region dfs each containing coefficient info
Output: Coefficient data df of specified lag time

corr_coef <- function(data) {
  place <- numeric(length=35)
  mobility <- numeric(length=35)
  coefficient <- numeric(length=35)
  p_val <- numeric(length=35)
  i=1
  j=1
  k=1
  for(loc in data){
    for(corr in loc) {
      place[i] <- names(data)[j]
      mobility[i] <- names(loc)[k]
      coefficient[i] <- corr$estimate
      p_val[i] <- corr$p.value
      i=i+1
      k=k+1
    }
    j=j+1
    k=1
  }
  df <- data.frame(place, mobility, coefficient, p_val)
  return(df)
}

get_regressions <- function(data, lag) {
  if(lag==0) {
    retail <- lm(Rt ~ retail + date, data = data)
    grocery <- lm(Rt ~ grocery + date, data = data)
    parks <- lm(Rt ~ parks + date, data = data)
    transit <- lm(Rt ~ transit + date, data = data)
    workplaces <- lm(Rt ~ workplaces + date, data = data)
    residential <- lm(Rt ~ residential + date, data = data)
    global <- lm(Rt ~ global + date, data = data)
  } else if(lag==7) {
    retail <- lm(Rt ~ retail_lag7 + date, data = data)
    grocery <- lm(Rt ~ grocery_lag7 + date, data = data)
    parks <- lm(Rt ~ parks_lag7 + date, data = data)
    transit <- lm(Rt ~ transit_lag7 + date, data = data)
    workplaces <- lm(Rt ~ workplaces_lag7 + date, data = data)
    residential <- lm(Rt ~ residential_lag7 + date, data = data)
    global <- lm(Rt ~ global_lag7 + date, data = data)
  } else if(lag==14) {
    retail <- lm(Rt ~ retail_lag14 + date, data = data)
    grocery <- lm(Rt ~ grocery_lag14 + date, data = data)
    parks <- lm(Rt ~ parks_lag14 + date, data = data)
    transit <- lm(Rt ~ transit_lag14 + date, data = data)
    workplaces <- lm(Rt ~ workplaces_lag14 + date, data = data)
    residential <- lm(Rt ~ residential_lag14 + date, data = data)
    global <- lm(Rt ~ global_lag14 + date, data = data)
  } else {
    stop('Must enter 0, 7, or 14')
  }
  return(list(Retail=retail, Grocery=grocery, Parks=parks, Transit=transit, Workplaces=workplaces, Residential=residential, Global=global))
}

Function to get regression coefficient

Input: List of region dfs each containing regression info
Output: Coefficient data df of specified lag time

reg_coef <- function(data) {
  place <- numeric(length=35)
  mobility<- numeric(length=35)
  coefficient <- numeric(length=35)
  r2 <- numeric(length=35)
  p_val <- numeric(length=35)
  i=1
  j=1
  k=1
  for(loc in data){
    for(reg in loc) {
      place[i] <- names(data)[j]
      mobility[i] <- names(loc)[k]
      coefficient[i] <- tidy(reg)[[2,2]]
      r2[i] <- glance(reg)[[1,2]]
      p_val[i] <- glance(reg)[[1,5]]
      i=i+1
      k=k+1
    }
    j=j+1
    k=1
  }
  df <- data.frame(place, mobility, coefficient, r2, p_val)
  return(df)
}
